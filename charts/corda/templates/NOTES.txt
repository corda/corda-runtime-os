kubectl get secret {{ include "corda.restApiAdminPasswordSecretName" . }} --namespace {{ .Release.Namespace }} -o go-template='{{ `{{` }} .data.{{ include "corda.restApiAdminSecretPasswordKey" . }} | base64decode {{ `}}` }}'
